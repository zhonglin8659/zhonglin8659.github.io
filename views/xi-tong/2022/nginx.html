    安装Nginx

    兜里有糖 2022/8/12 Linux

    # ubuntu安装nginx

    lsb_release -a
    1

    # 安装环境

    # 更新apt源

    apt update
    1

    # 安装gcc，g++开发类库

    apt install build-essential
    1
    apt install libtool
    1

    # 安装pcre库

    apt install libpcre3 libpcre3-dev
    1

    # 安装zlib库

    apt install zlib1g-dev
    1

    # 安装oepnssl库

    apt install openssl libssl-dev
    1

    # 安装nginx

    # 下载nginx包

    wget http://nginx.org/download/nginx-1.23.0.tar.gz
    1

    # 解压文件

    tar -zxvf nginx-1.23.0.tar.gz
    1

    # 切换到nginx目录

    cd nginx-1.23.0
    1

    # 预编译

    ./configure
    1

    # 安装

    make && make install
    1

    # 启动nginx

    cd /usr/local/nginx/sbin
./nginx
    1
